def self.from_open_struct(open_struct)
  order_tracking = OrderTracking.new
  order_tracking.state = open_struct.state
  unless open_struct.driver.nil?
    order_tracking.driver = Driver.from_open_struct(open_struct.driver)
  end
  order_tracking.pickup_date = open_struct.pickupDate
  order_tracking.estimated_delivery_date = open_struct.estimatedDeliveryDate
  return order_tracking
end
