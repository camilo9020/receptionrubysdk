def self.build_from_response(response)
  code = !response.status_code.to_s.empty? ? response.status_code : 'INTERNAL_SERVER_ERROR'
  msg = response.body ? response.body.to_s : 'Internal server error'
  api_exception = ApiException.new(msg, code)
  return api_exception
end
