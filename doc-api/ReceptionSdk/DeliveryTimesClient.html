def get_all
  begin
    request = Request.new
    request.endpoint = "deliveryTimes"
    delivery_times = []
    response = @connection.get(request)
    if response.status_code == HttpStatus::OK

      delivery_times_json = JsonUtil::adapt_response_for_json_parsing(response)

      # FIXME : Find a way of Mapping the JSON result into a model directly
      delivery_times_response = DeliveryTimesResponse.from_open_struct(JSON.parse(delivery_times_json, object_class: OpenStruct))
      if delivery_times_response.delivery_times.length == delivery_times_response.total
        return delivery_times_response.delivery_times
      else
        delivery_times += delivery_times_response.delivery_times

        while !delivery_times_response.delivery_times.nil? && delivery_times_response.delivery_times.length != 0
          request.endpoint = "deliveryTimes?offset=#{delivery_times.length}"
          response = @connection.get(request)
          if response.status_code == HttpStatus::OK
            delivery_times_json = JsonUtil::adapt_response_for_json_parsing(response)
            delivery_times_response = DeliveryTimesResponse.from_open_struct(JSON.parse(delivery_times_json, object_class: OpenStruct))

            delivery_times += delivery_times_response.delivery_times
          else
            raise ApiException.build_from_response(response)
          end
        end

        return delivery_times
      end
    else
      raise ApiException.build_from_response(response)
    end
  rescue ApiException => e
    raise e
  rescue StandardError => se
    raise ApiException.new(se.message)
  end
end
