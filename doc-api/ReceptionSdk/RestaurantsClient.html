def close(restaurant, from, to, reason)
  Ensure.argument_not_null(restaurant, "restaurant")
  Ensure.argument_not_null(from, "from")
  Ensure.argument_not_null(to, "to")
  Ensure.argument_not_null(reason, "reason")
  Ensure.first_date_before_second(from, to, "from", "to")

  if restaurant.is_a? Numeric
    Ensure.greater_than_zero(restaurant, "restaurant")
    update(restaurant, from, to, reason)
  else
    update(restaurant.id, from, to, reason)
  end

end

def get_all(pagination_options, integration_code = nil)
  begin
    request = Request.new
    request.endpoint = "restaurants?offset=#{pagination_options.offset}&limit=#{pagination_options.limit}"

    unless integration_code.nil? || integration_code.empty?
      request.endpoint += "&code=#{integration_code}"
    end

    response = @connection.get(request)
    if response.status_code == HttpStatus.OK
      return JSON.parse(response.body.to_json, object_class: RestaurantsResponse).restaurants
    else
      raise ApiException.new(response)
    end
  rescue ApiException => e
    raise e
  rescue StandardError => se
    raise ApiException.new(se.response)
  end
end

def open(restaurant, from)
  Ensure.argument_not_null(restaurant, "restaurant")
  Ensure.argument_not_null(from, "from")

  if restaurant.is_a? Numeric
    Ensure.greater_than_zero(restaurant, "restaurant")
    update(restaurant, from, nil, nil)
  else
    update(restaurant.id, from, nil, nil)
  end
end

def update(restaurant_id, from, to, reason)
  begin
    request = Request.new
    request.endpoint = "restaurants/#{restaurant_id}"

    body = Hash.new
    if to.nil?
      body["state"] = RestaurantState.ONLINE
      body["from"] = DateUtil.format_date(from)
    else
      body["state"] = RestaurantState.OFFLINE
      body["from"] = DateUtil.format_date(from)
      body["to"] = DateUtil.format_date(to)
      body["reason"] = reason
    end
    request.body = body

    response = @connection.put(request)
    if response.status_code != HttpStatus.OK
      raise ApiException.new(response)
    end
  rescue ApiException => e
    raise e
  rescue StandardError => se
    raise ApiException.new(se.response)
  end
end
