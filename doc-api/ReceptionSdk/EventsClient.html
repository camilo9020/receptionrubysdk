def acknowledgement(order_id, restaurant = nil)
  if @connection.credentials.centralized_keys && restaurant.nil?
    raise ArgumentError.new("You must specify a restaurant_code or restaurant_id")
  end

  if restaurant.is_a? String
    Ensure.argument_not_null_or_empty_string(restaurant, "restaurant_code")
    acknowledgement_event(order_id, restaurant)
  elsif restaurant.is_a? Numeric
    Ensure.greater_than_zero(restaurant, "restaurant_id")
    acknowledgement_event(order_id, nil, restaurant)
  else
    acknowledgement_event(order_id)
  end
end

def initialization(version, restaurant = nil)
  if @connection.credentials.centralized_keys && restaurant.nil?
    raise ArgumentError.new("You must specify a restaurant_code or restaurant_id")
  end
  Ensure.argument_not_null_or_empty_map(version, "version")

  if restaurant.is_a? String
    Ensure.argument_not_null_or_empty_string(restaurant, "restaurant_code")

    initialization_event(version, restaurant)
  elsif restaurant.is_a? Numeric
    Ensure.greater_than_zero(restaurant, "restaurant_id")
    initialization_event(version, nil, restaurant)
  else
    initialization_event(version)
  end

end
