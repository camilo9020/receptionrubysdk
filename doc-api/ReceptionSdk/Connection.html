def get(request)
  begin
    request.headers["Accept"] = :json
    request.headers["Content-Type"] = :json

    api_response = RestClient::Request.execute(method: :get,url: @base_url + request.endpoint, payload: request.body.to_json,
                                      headers: request.headers, timeout: request.timeout)

    response = Response.new
    response.status_code = api_response.code
    response.body = api_response.body
    response.content = api_response.body
    return response
  rescue RestClient::Forbidden => fe
    puts "Catching exception for forbidden access to api"
    response = Response.new
    response.status_code = HttpStatus::FORBIDDEN
    response.body = "FORBIDDEN"
    return response
  rescue RestClient::ExceptionWithResponse => ewr
    response = Response.new
    response.status_code = ewr.response.code
    response.body = ewr.response.body
    puts "Error : #{response.status_code} #{response.body.to_json}"
    response.content = ewr.response.body
    return response
  rescue StandardError => e
    raise ApiException.new("Cannot execute operation, an unknown error has occurred : #{e.message}")
  end
end
