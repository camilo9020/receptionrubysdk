def get_all
  begin
    request = Request.new
    request.endpoint = "rejectMessages"

    reject_messages = []

    response = @connection.get(request)
    if response.status_code == HttpStatus::OK
      reject_messages_json = JsonUtil::adapt_response_for_json_parsing(response)

      # FIXME : Find a way of Mapping the JSON result into a model directly
      reject_messages_response = RejectMessagesResponse.from_open_struct(JSON.parse(reject_messages_json, object_class: OpenStruct))

      if reject_messages_response.reject_messages.length == reject_messages_response.total
        return reject_messages_response.reject_messages
      else
        reject_messages += reject_messages_response.reject_messages

        while !(reject_messages_response.reject_messages.nil?) && reject_messages_response.reject_messages.length != 0
          request.endpoint = "rejectMessages?offset=#{reject_messages.length}"

          response = @connection.get(request)
          if response.status_code == HttpStatus::OK
            reject_messages_json = JsonUtil::adapt_response_for_json_parsing(response)
            reject_messages_response = RejectMessagesResponse.from_open_struct(JSON.parse(reject_messages_json, object_class: OpenStruct))
            reject_messages += reject_messages_response.reject_messages
          else
            raise ApiException.new(response)
          end

        end
        
        return reject_messages
      end
    else
      raise ApiException.new(response)
    end
  rescue ApiException => e
    raise e
  rescue StandardError => se
    raise ApiException.new(se.message)
  end
end
