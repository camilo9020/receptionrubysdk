def self.argument_not_null_or_empty_map(value, name)
  argument_not_null(value, name)
  if (value.nil? || value.empty?)
    raise(ArgumentError, "Argument #{name} cannot be empty")
  end
end

def self.argument_not_null_or_empty_string(value, name)
  argument_not_null(value, name)
  if (value.to_s.empty? || value.strip.to_s.empty?)
    raise(ArgumentError, "Argument #{name} cannot be empty")
  end
end

def self.first_date_before_second(first_date, second_date, first_date_name, second_date_name)
  if (first_date.eql?(second_date) || first_date > second_date)
    raise(ArgumentError, "Argument #{first_date_name} must be before than #{second_date_name}")
  end
end
